## 🛠️ PPT 제작 방법

### ⚠️ AI로 PPT 만들기 (권장 방법)

언어 모델(ChatGPT, Gemini 등)은 직접 PPT 파일을 생성할 수 없습니다.
아래 방법 중 하나를 선택하세요:

#### 방법 1: Gamma.app (가장 쉬움) ⭐ 추천
1. https://gamma.app 접속
2. "Create new" → "Paste in text"
3. 이 문서의 내용을 복사해서 붙여넣기
4. 자동으로 PPT 스타일 슬라이드 생성
5. 이미지 위치에 스크린샷 직접 업로드
6. PPT/PDF로 내보내기

#### 방법 2: Canva
1. https://canva.com 접속
2. "프레젠테이션" 템플릿 선택
3. 각 슬라이드 내용을 수동으로 입력
4. 스크린샷 업로드 및 배치
5. PPT/PDF로 내보내기

#### 방법 3: Python 스크립트 (자동화)
아래 스크립트를 실행하면 자동으로 PPT 파일 생성:

```python
# ppt/generate_ppt.py 파일로 저장 후 실행
# pip install python-pptx Pillow

from pptx import Presentation
from pptx.util import Inches, Pt
from pptx.dml.color import RgbColor
from pptx.enum.text import PP_ALIGN
import os

# PPT 생성
prs = Presentation()
prs.slide_width = Inches(13.33)  # 16:9
prs.slide_height = Inches(7.5)

# 색상 정의
PRIMARY_COLOR = RgbColor(99, 102, 241)  # #6366F1

def add_title_slide(prs, title, subtitle, logo_path=None):
    slide = prs.slides.add_slide(prs.slide_layouts[6])  # 빈 슬라이드
    
    # 로고
    if logo_path and os.path.exists(logo_path):
        slide.shapes.add_picture(logo_path, Inches(5.67), Inches(1), Inches(2), Inches(2))
    
    # 제목
    title_box = slide.shapes.add_textbox(Inches(1), Inches(3.5), Inches(11.33), Inches(1))
    tf = title_box.text_frame
    p = tf.paragraphs[0]
    p.text = title
    p.font.size = Pt(60)
    p.font.bold = True
    p.alignment = PP_ALIGN.CENTER
    
    # 부제목
    sub_box = slide.shapes.add_textbox(Inches(1), Inches(4.5), Inches(11.33), Inches(0.5))
    tf = sub_box.text_frame
    p = tf.paragraphs[0]
    p.text = subtitle
    p.font.size = Pt(32)
    p.alignment = PP_ALIGN.CENTER
    
    return slide

def add_content_slide(prs, title, bullets):
    slide = prs.slides.add_slide(prs.slide_layouts[6])
    
    # 제목
    title_box = slide.shapes.add_textbox(Inches(0.5), Inches(0.3), Inches(12), Inches(0.8))
    tf = title_box.text_frame
    p = tf.paragraphs[0]
    p.text = title
    p.font.size = Pt(44)
    p.font.bold = True
    
    # 불릿 포인트
    content_box = slide.shapes.add_textbox(Inches(0.5), Inches(1.5), Inches(12), Inches(5))
    tf = content_box.text_frame
    for i, bullet in enumerate(bullets):
        if i == 0:
            p = tf.paragraphs[0]
        else:
            p = tf.add_paragraph()
        p.text = f"• {bullet}"
        p.font.size = Pt(24)
        p.space_after = Pt(12)
    
    return slide

def add_image_slide(prs, title, image_path, bullets):
    slide = prs.slides.add_slide(prs.slide_layouts[6])
    
    # 제목
    title_box = slide.shapes.add_textbox(Inches(0.5), Inches(0.3), Inches(12), Inches(0.8))
    tf = title_box.text_frame
    p = tf.paragraphs[0]
    p.text = title
    p.font.size = Pt(44)
    p.font.bold = True
    
    # 이미지 (왼쪽)
    if image_path and os.path.exists(image_path):
        slide.shapes.add_picture(image_path, Inches(0.5), Inches(1.3), Inches(6), Inches(5.5))
    else:
        # 이미지 플레이스홀더
        placeholder = slide.shapes.add_textbox(Inches(0.5), Inches(1.3), Inches(6), Inches(5.5))
        tf = placeholder.text_frame
        p = tf.paragraphs[0]
        p.text = f"[스크린샷 삽입]\\n{image_path}"
        p.font.size = Pt(18)
        p.alignment = PP_ALIGN.CENTER
    
    # 설명 (오른쪽)
    content_box = slide.shapes.add_textbox(Inches(7), Inches(1.5), Inches(5.5), Inches(5))
    tf = content_box.text_frame
    for i, bullet in enumerate(bullets):
        if i == 0:
            p = tf.paragraphs[0]
        else:
            p = tf.add_paragraph()
        p.text = bullet
        p.font.size = Pt(20)
        p.space_after = Pt(8)
    
    return slide

# === 슬라이드 생성 ===

# 01. 표지
add_title_slide(prs, "CoUp (코업)", "스터디 협업 플랫폼", "../mainlogo.png")

# 02. 목차
add_content_slide(prs, "📋 목차", [
    "01  프로젝트 소개",
    "02  문제 인식 & 해결 방안",
    "03  주요 기능 (8개)",
    "04  기술 스택 & 아키텍처",
    "05  데모 영상",
    "06  개발 성과 & 마무리"
])

# 03. 프로젝트 소개
add_content_slide(prs, "📊 프로젝트 소개", [
    "CoUp (코업) = Co(operate) + Up = 함께 성장",
    "",
    "💡 스터디 탐색부터 협업까지, 올인원 스터디 플랫폼",
    "",
    "📅 개발 기간: 약 6주 (2025.10.28 ~ 12.07)",
    "👤 인원: 1인 풀스택",
    "🛠️ 주요 기술: Next.js 16, React 19"
])

# ... 나머지 슬라이드들도 동일하게 추가 ...

# PPT 저장
prs.save("CoUp_Presentation.pptx")
print("✅ PPT 생성 완료: CoUp_Presentation.pptx")
```
