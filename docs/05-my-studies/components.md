export function getStudyHeaderStyle(studyId) {
  const hash = studyId.split('').reduce((acc, char) => {
    return char.charCodeAt(0) + ((acc << 5) - acc)
  }, 0)
  
  const hue = Math.abs(hash % 360)

    background: `linear-gradient(135deg, hsl(${hue}, 70%, 50%), hsl(${hue + 30}, 60%, 40%))`,
    color: 'white'
  }
}
