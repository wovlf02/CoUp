  const [localStream, setLocalStream] = useState<MediaStream | null>(null)
  const [remoteStreams, setRemoteStreams] = useState<Map<string, MediaStream>>(new Map())
  const [peers, setPeers] = useState<Map<string, RTCPeerConnection>>(new Map())
  const { socket } = useSocket()
  
  // ë¡œì»¬ ë¯¸ë””ì–´ ìŠ¤íŠ¸ë¦¼ ì´ˆê¸°í™”
  const initLocalStream = async () => {
    try {
      const stream = await navigator.mediaDevices.getUserMedia({
        video: {
          width: { ideal: 1280 },
          height: { ideal: 720 },
          frameRate: { ideal: 30 }
        },
        audio: {
          echoCancellation: true,
          noiseSuppression: true,
          autoGainControl: true
        }
      })
      
      setLocalStream(stream)
      return stream
    } catch (error) {
      console.error('Failed to get local stream:', error)
      throw error
    }
  }
  
  // ë°© ì…ì¥
  const joinRoom = async () => {
    const stream = await initLocalStream()
    
    // Socket.ioë¡œ ë°© ì…ì¥ ì•Œë¦¼
    socket.emit('join-video-room', {
      studyId,
      roomId,
      userId: currentUser.id
    })
    
    // ê¸°ì¡´ ì°¸ì—¬ìë“¤ì˜ offer ë°›ê¸°
    socket.on('user-joined', async ({ userId, offer }) => {
      const peer = createPeerConnection(userId, stream)
      await peer.setRemoteDescription(new RTCSessionDescription(offer))
      
      const answer = await peer.createAnswer()
      await peer.setLocalDescription(answer)
      
      socket.emit('answer', { to: userId, answer })
    })
    
    // ìƒˆ ì°¸ì—¬ìì˜ answer ë°›ê¸°
    socket.on('answer', async ({ from, answer }) => {
      const peer = peers.get(from)
      if (peer) {
        await peer.setRemoteDescription(new RTCSessionDescription(answer))
      }
    })
    
    // ICE candidate êµí™˜
    socket.on('ice-candidate', async ({ from, candidate }) => {
      const peer = peers.get(from)
      if (peer) {
        await peer.addIceCandidate(new RTCIceCandidate(candidate))
      }
    })
  }
  
  // Peer Connection ìƒì„±
  const createPeerConnection = (userId: string, localStream: MediaStream) => {
    const peer = new RTCPeerConnection({
      iceServers: [
        { urls: 'stun:stun.l.google.com:19302' },
        { urls: 'stun:stun1.l.google.com:19302' }
      ]
    })
    
    // ë¡œì»¬ ìŠ¤íŠ¸ë¦¼ íŠ¸ë™ ì¶”ê°€
    localStream.getTracks().forEach(track => {
      peer.addTrack(track, localStream)
    })
    
    // ì›ê²© ìŠ¤íŠ¸ë¦¼ ìˆ˜ì‹ 
    peer.ontrack = (event) => {
      const [remoteStream] = event.streams
      setRemoteStreams(prev => new Map(prev).set(userId, remoteStream))
    }
    
    // ICE candidate ì „ì†¡
    peer.onicecandidate = (event) => {
      if (event.candidate) {
        socket.emit('ice-candidate', {
          to: userId,
          candidate: event.candidate
        })
      }
    }
    
    setPeers(prev => new Map(prev).set(userId, peer))
    return peer
  }
  
  // ìŒì†Œê±° í† ê¸€
  const toggleMute = () => {
    if (localStream) {
      const audioTrack = localStream.getAudioTracks()[0]
      audioTrack.enabled = !audioTrack.enabled
      return audioTrack.enabled
    }
  }
  
  // ë¹„ë””ì˜¤ í† ê¸€
  const toggleVideo = () => {
    if (localStream) {
      const videoTrack = localStream.getVideoTracks()[0]
      videoTrack.enabled = !videoTrack.enabled
      return videoTrack.enabled
    }
  }
  
  // í™”ë©´ ê³µìœ 
  const shareScreen = async () => {
    try {
      const screenStream = await navigator.mediaDevices.getDisplayMedia({
        video: {
          cursor: 'always'
        },
        audio: false
      })
      
      // ë¹„ë””ì˜¤ íŠ¸ë™ì„ í™”ë©´ ê³µìœ ë¡œ êµì²´
      const screenTrack = screenStream.getVideoTracks()[0]
      
      peers.forEach(peer => {
        const sender = peer.getSenders().find(s => s.track?.kind === 'video')
        if (sender) {
          sender.replaceTrack(screenTrack)
        }
      })
      
      // í™”ë©´ ê³µìœ  ì¢…ë£Œ ì‹œ ì›ë˜ ë¹„ë””ì˜¤ë¡œ ë³µêµ¬
      screenTrack.onended = () => {
        const videoTrack = localStream?.getVideoTracks()[0]
        if (videoTrack) {
          peers.forEach(peer => {
            const sender = peer.getSenders().find(s => s.track?.kind === 'video')
            if (sender) {
              sender.replaceTrack(videoTrack)
            }
          })
        }
      }
      
      return screenStream
    } catch (error) {
      console.error('Failed to share screen:', error)
      throw error
    }
  }
  
  // ë°© ë‚˜ê°€ê¸°
  const leaveRoom = () => {
    // ë¡œì»¬ ìŠ¤íŠ¸ë¦¼ ì¢…ë£Œ
    localStream?.getTracks().forEach(track => track.stop())
    
    // ëª¨ë“  Peer Connection ì¢…ë£Œ
    peers.forEach(peer => peer.close())
    
    // Socket ì•Œë¦¼
    socket.emit('leave-video-room', { studyId, roomId })
    
    // ìƒíƒœ ì´ˆê¸°í™”
    setLocalStream(null)
    setRemoteStreams(new Map())
    setPeers(new Map())
  }
  
  return {
    localStream,
    remoteStreams,
    joinRoom,
    leaveRoom,
    toggleMute,
    toggleVideo,
    shareScreen
  }
}
