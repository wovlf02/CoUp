const handleCreateTask = async (taskData) => {
  // 유효성 검증
  if (!taskData.title || taskData.title.length < 2) {
    toast.error('제목은 2자 이상 입력해주세요')
    return
  }
  
  try {
    setIsSubmitting(true)
    
    const task = await api.post(`/api/v1/my-studies/${studyId}/tasks`, {
      title: taskData.title,
      description: taskData.description,
      status: taskData.status || 'TODO',
      priority: taskData.priority || 'MEDIUM',
      assignees: taskData.assignees,
      dueDate: taskData.dueDate,
      labels: taskData.labels,
      checklist: taskData.checklist,
      attachments: taskData.attachments
    })
    
    toast.success('할일이 추가되었습니다')
    
    // 담당자에게 알림
    if (taskData.assignees.length > 0) {
      await api.post(`/api/v1/notifications/task-assigned`, {
        taskId: task.id,
        assignees: taskData.assignees
      })
    }
    
    // 목록 새로고침
    queryClient.invalidateQueries(['tasks', studyId])
    
    closeModal()
  } catch (error) {
    toast.error('할일 추가 중 오류가 발생했습니다')
  } finally {
    setIsSubmitting(false)
  }
}

### 2. 드래그 앤 드롭으로 상태 변경

```javascript
const handleDragEnd = async (result) => {
  const { destination, source, draggableId } = result
  
  // 드롭 위치가 없거나 같은 위치면 무시
  if (!destination || 
      (destination.droppableId === source.droppableId && 
       destination.index === source.index)) {
    return
  }
  
  const taskId = draggableId
  const newStatus = destination.droppableId // 'TODO', 'IN_PROGRESS', 'DONE', 'ON_HOLD'
  
  try {
    // Optimistic Update
    const previousTasks = queryClient.getQueryData(['tasks', studyId])
    
    // 상태 즉시 변경 (UI)
    queryClient.setQueryData(['tasks', studyId], (old) => {
      return old.map(task => 
        task.id === taskId 
          ? { ...task, status: newStatus }
          : task
      )
    })
    
    // 서버에 업데이트
    await api.patch(`/api/v1/my-studies/${studyId}/tasks/${taskId}/status`, {
      status: newStatus
    })
    
    toast.success('상태가 변경되었습니다')
    
    // 완료 상태로 변경 시 축하 메시지
    if (newStatus === 'DONE') {
      toast.success('🎉 할일을 완료했습니다!')
    }
  } catch (error) {
    // 실패 시 롤백
    queryClient.setQueryData(['tasks', studyId], previousTasks)
    toast.error('상태 변경 중 오류가 발생했습니다')
  }
}
```

### 4. 마감일 임박 알림

```javascript
const checkUpcomingDeadlines = () => {
  const tasks = queryClient.getQueryData(['tasks', studyId])
  const now = new Date()
  
  tasks.forEach(task => {
    if (task.dueDate && task.status !== 'DONE') {
      const dueDate = new Date(task.dueDate)
      const daysUntil = Math.ceil((dueDate - now) / (1000 * 60 * 60 * 24))
      
      // D-1 알림
      if (daysUntil === 1) {
        toast.warning(`⚠️ "${task.title}" 마감까지 1일 남았습니다!`)
      }
      
      // 마감일 지남
      if (daysUntil < 0) {
        toast.error(`🔴 "${task.title}" 마감일이 지났습니다!`)
      }
    }
  })
}

// 페이지 로드 시 & 5분마다 체크
useEffect(() => {
  checkUpcomingDeadlines()
  const interval = setInterval(checkUpcomingDeadlines, 5 * 60 * 1000)
  return () => clearInterval(interval)
}, [])
```
