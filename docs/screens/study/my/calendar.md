# 캘린더 (Study Calendar)

> **화면 ID**: `STUDY-CALENDAR-01`  
> **라우트**: `/my-studies/[studyId]/calendar`  
> **목적**: 스터디 일정 관리 및 시각화  
> **접근 권한**: MEMBER+ (읽기), MEMBER+ (일정 추가), ADMIN+ (수정/삭제)  
> **렌더링**: CSR (실시간 업데이트)

---

## 🎯 화면 목적

**"스터디 일정을 한눈에"**
- 월/주/일 뷰로 일정 확인
- 일정 추가/수정/삭제
- 구글 캘린더 연동
- 다가오는 일정 알림

```javascript
const handleCreateEvent = async (eventData) => {
  // 유효성 검증
  if (!eventData.title || eventData.title.length < 2) {
    toast.error('제목은 2자 이상 입력해주세요')
    return
  }
  
  if (!eventData.startDate || !eventData.endDate) {
    toast.error('날짜와 시간을 선택해주세요')
    return
  }
  
  if (new Date(eventData.startDate) >= new Date(eventData.endDate)) {
    toast.error('종료 시간이 시작 시간보다 빨라야 합니다')
    return
  }
  
  try {
    setIsSubmitting(true)
    
    const event = await api.post(`/api/v1/my-studies/${studyId}/calendar`, {
      title: eventData.title,
      startDate: eventData.startDate,
      endDate: eventData.endDate,
      location: eventData.location,
      description: eventData.description,
      attendees: eventData.attendees,
      category: eventData.category,
      color: eventData.color,
      isRecurring: eventData.isRecurring,
      recurrenceRule: eventData.recurrenceRule,
      reminders: eventData.reminders
    })
    
    toast.success('일정이 추가되었습니다')
    
    // 참석자에게 알림
    if (eventData.attendees.length > 0) {
      await api.post(`/api/v1/notifications/calendar-invite`, {
        eventId: event.id,
        attendees: eventData.attendees
      })
    }
    
    // 캘린더 새로고침
    queryClient.invalidateQueries(['calendar', studyId])
    
    closeModal()
  } catch (error) {
    toast.error('일정 추가 중 오류가 발생했습니다')
  } finally {
    setIsSubmitting(false)
  }
}
```

```javascript
const handleDeleteEvent = async (eventId, isRecurring) => {
  let confirmed = false
  
  if (isRecurring) {
    const choice = confirm(
      '반복 일정입니다. 어떻게 삭제하시겠습니까?\n\n' +
      '[확인] 이 일정만 삭제\n' +
      '[취소] 모든 반복 일정 삭제'
    )
    
    if (choice === null) return // 취소
    
    confirmed = choice
  } else {
    confirmed = confirm('정말 삭제하시겠습니까?')
  }
  
  if (!confirmed) return
  
  try {
    await api.delete(`/api/v1/my-studies/${studyId}/calendar/${eventId}`, {
      data: { deleteAll: !isRecurring }
    })
    
    toast.success('일정이 삭제되었습니다')
    
    queryClient.invalidateQueries(['calendar', studyId])
  } catch (error) {
    toast.error('삭제 중 오류가 발생했습니다')
  }
}
```

```javascript
const handleGoogleCalendarSync = async () => {
  try {
    // OAuth 인증
    const authUrl = await api.get(`/api/v1/my-studies/${studyId}/calendar/google-auth`)
    
    // 팝업으로 인증 페이지 열기
    const popup = window.open(authUrl, 'Google Calendar Auth', 'width=600,height=700')
    
    // 인증 완료 대기
    const interval = setInterval(async () => {
      if (popup.closed) {
        clearInterval(interval)
        
        // 연동 상태 확인
        const isConnected = await checkGoogleCalendarConnection()
        
        if (isConnected) {
          toast.success('구글 캘린더와 연동되었습니다')

          await api.post(`/api/v1/my-studies/${studyId}/calendar/sync-google`)
          
          queryClient.invalidateQueries(['calendar', studyId])
        }
      }
    }, 1000)
  } catch (error) {
    toast.error('연동 중 오류가 발생했습니다')
  }
}
```
